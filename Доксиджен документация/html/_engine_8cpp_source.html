53void Engine::removeSpaces(std::string& string)
54{
55 std::size_t currentIndex = 0;
56 std::size_t length = string.length();
57
58 std::string result = "";
59
60 while (string[currentIndex] == ' ')
61 {
62 ++currentIndex;
63 }
64
65 bool spaceFound = false;
66
67 while (currentIndex < length)
68 {
69 if (string[currentIndex] != ' ')
70 {
71 result.push_back(string[currentIndex++]);
72 spaceFound = false;
73 }
74 else
75 {
76 if (!spaceFound)
77 {
78 result.push_back(string[currentIndex++]);
79 spaceFound = true;
80 }
81 else
82 {
83 ++currentIndex;
84 }
85 }
86 }
87
88 if (result.back() == ' ')
89 {
90 result.pop_back();
91 }
92
93 string = result;
